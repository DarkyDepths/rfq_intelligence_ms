03 WorkbookReader — The Format Adapter

3.1 Why this exists

Every extractor calls reader.get_cell(sheet, row, col) — never xlrd directly. When .xlsx support is needed later, you write an XlsxReader that implements the same interface. Zero extractor changes.

3.2 Interface contract

# parsers/workbook/workbook_reader.py
from abc import ABC, abstractmethod
from typing import Any, List, Tuple, Optional
from datetime import date

class WorkbookReader(ABC):

    @abstractmethod
    def open(self, file_path: str) -> None:
        """Open workbook file. Raises WorkbookOpenError on failure."""

    @abstractmethod
    def sheet_names(self) -> List[str]:
        """Return all sheet names in workbook order."""

    @abstractmethod
    def has_sheet(self, name: str) -> bool:
        """Check if sheet exists."""

    @abstractmethod
    def get_cell_value(self, sheet: str, row: int, col: int) -> Any:
        """Return cell value. row/col are 0-indexed."""

    @abstractmethod
    def get_cell_type(self, sheet: str, row: int, col: int) -> str:
        """Return normalized type: 'empty','text','number','date','boolean','error','blank'"""

    @abstractmethod
    def get_date_value(self, sheet: str, row: int, col: int) -> Optional[date]:
        """Return Python date from Excel serial. Uses correct datemode. Returns None on failure."""

    @abstractmethod
    def get_merged_regions(self, sheet: str) -> List[Tuple[int, int, int, int]]:
        """Return list of (row_lo, row_hi, col_lo, col_hi) — 0-indexed."""

    @abstractmethod
    def get_sheet_dimensions(self, sheet: str) -> Tuple[int, int]:
        """Return (nrows, ncols)."""

    @abstractmethod
    def close(self) -> None:
        """Release resources."""

3.3 XlsReader — the V1 implementation

# parsers/workbook/xls_reader.py
import xlrd
from datetime import date

class XlsReader(WorkbookReader):

    def __init__(self):
        self._wb = None
        self._sheets = {}

    def open(self, file_path: str):
        self._wb = xlrd.open_workbook(file_path, formatting_info=True)
        # Cache sheet objects by name
        for name in self._wb.sheet_names():
            self._sheets[name] = self._wb.sheet_by_name(name)

    def get_cell_type(self, sheet, row, col):
        TYPE_MAP = {0:'empty', 1:'text', 2:'number', 3:'date',
                    4:'boolean', 5:'error', 6:'blank'}
        return TYPE_MAP[self._sheets[sheet].cell_type(row, col)]

    def get_date_value(self, sheet, row, col):
        val = self._sheets[sheet].cell_value(row, col)
        ctype = self._sheets[sheet].cell_type(row, col)
        if ctype != 3:
            return None
        try:
            dt = xlrd.xldate_as_tuple(val, self._wb.datemode)
            return date(*dt[:3])
        except:
            return None

    # ... remaining methods wrap xlrd calls similarly
Key implementation note
formatting_info=True is required in xlrd.open_workbook() to access merged cell regions. This is needed for Bid S (50 merges) and General (24 merges). Without it, merged cell data is unavailable.

06 Assembler — Building the Envelope

The assembler is a pure function that takes the three ExtractionResult objects and produces the complete WorkbookParseEnvelope. It contains no parsing logic — only structural composition.

6.1 Complete envelope assembly v1.1 fix — schema-aligned

# parsers/workbook/assembler.py
from datetime import datetime

# Template identifier — neutral, no sheet count encoded
TEMPLATE_NAME = "ghi_estimation_workbook_v1"
PARSER_VERSION = "workbook-parser-v1.1"

def assemble_envelope(
    rfq_id: str,
    file_name: str,
    blob_path: str,
    general_result: ExtractionResult,
    bid_s_result: ExtractionResult,
    top_sheet_result: ExtractionResult,
    cross_checks: List[dict],
) -> dict:

    # Collect all issues across sheets
    all_issues = general_result.issues + bid_s_result.issues + top_sheet_result.issues
    all_warnings = [i for i in all_issues if i["severity"] == "warning"]
    all_errors = [i for i in all_issues if i["severity"] == "error"]

    # Determine overall status
    if all_errors or not all(r.template_match for r in [general_result, bid_s_result, top_sheet_result]):
        status = "failed"
    elif all_warnings:
        status = "parsed_with_warnings"
    else:
        status = "parsed_ok"

    return {
        # ── Envelope metadata (all schema-required fields present) ──
        "rfq_id": rfq_id,
        "template_name": TEMPLATE_NAME,
        "workbook_format": "xls",
        "workbook_file_name": file_name,
        "workbook_blob_path": blob_path,
        "template_match": all(r.template_match for r in [general_result, bid_s_result, top_sheet_result]),
        "parsed_at": datetime.utcnow().isoformat() + "Z",
        "parser_version": PARSER_VERSION,

        # ── workbook_profile ──
        "workbook_profile": {
            "rfq_identity": general_result.data["rfq_identity"],
            "identity_mirrors": {
                "bid_s": bid_s_result.data["identity_mirror"],
                "top_sheet": top_sheet_result.data["identity_mirror"],
            },
            "general_item_rows": general_result.data["general_item_rows"],
            "general_summary": general_result.data["general_summary"],
            "bid_meta": bid_s_result.data["bid_meta"],
        },

        # ── cost_breakdown_profile ──
        "cost_breakdown_profile": {
            "bid_summary_lines": bid_s_result.data["bid_summary_lines"],
            "bid_summary": bid_s_result.data["bid_summary"],
            "top_sheet_lines": top_sheet_result.data["top_sheet_lines"],
            "top_sheet_summary": top_sheet_result.data["top_sheet_summary"],
        },

        # ── parser_report ──
        "parser_report": {
            "status": status,
            "parsed_sheets": ["General", "Bid S", "Top Sheet"],
            "skipped_sheets": [],
            "warnings": all_warnings,
            "errors": all_errors,
            "anchor_checks": (general_result.anchor_checks
                              + bid_s_result.anchor_checks
                              + top_sheet_result.anchor_checks),

            "sheet_reports": {

08 Controller — The Orchestrator

Following BACAB's golden rule: the controller contains all business logic and orchestration. It never knows about HTTP. It calls datasources for data access and the parser module for domain logic. It returns the raw envelope dict — the translator handles shaping it for the API.

# controllers/workbook_parse_controller.py

class WorkbookParseController:

    def __init__(self, blob_datasource, snapshot_datasource):
        self.blob_ds = blob_datasource
        self.snapshot_ds = snapshot_datasource

    def parse_workbook(self, rfq_id: str, blob_path: str, file_name: str) -> dict:
        """
        Full orchestration:
        1. Download workbook from blob storage
        2. Open with WorkbookReader
        3. Run template validation
        4. Run all 3 extractors (General → Bid S → Top Sheet)
        5. Run cross-checks
        6. Assemble final envelope
        7. Store snapshot
        8. Return raw envelope dict
        """

        # Step 1: Download
        local_path = self.blob_ds.download_workbook(blob_path)

        # Step 2: Open reader
        reader = XlsReader()
        reader.open(local_path)

        try:
            # Step 3: Template check
            matcher = TemplateMatcher(reader)
            if not matcher.is_ghi_template():
                return self._build_failed_envelope(rfq_id, file_name, blob_path, "Template mismatch")

            # Step 4: Extract — strict order: General first (primary identity)
            general_result = GeneralExtractor(reader).extract()
            bid_s_result = BidSExtractor(reader).extract()
            top_sheet_result = TopSheetExtractor(reader).extract()

            # Step 5: Cross-checks
            cross_checks = run_cross_checks(
                general_result.data, bid_s_result.data, top_sheet_result.data
            )

            # Step 6: Assemble
            envelope = assemble_envelope(
                rfq_id, file_name, blob_path,
                general_result, bid_s_result, top_sheet_result,
                cross_checks,
            )

            # Step 7: Store
            self.snapshot_ds.save_parse_result(rfq_id, envelope)

            # Step 8: Return raw envelope (translator shapes this for HTTP)
            return envelope

        finally:
            reader.close()
