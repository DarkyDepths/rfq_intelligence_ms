04 Soft-Fail Contract — Three Distinct Outcomes

Missing ≠ Failed — these must be distinguishable in parser_report
None = sheet absent from workbook → sheet report is "skipped", artifact field is null.
Failed result = sheet present but extraction crashed → sheet report is "failed" with the error captured in issues, artifact field is null, error is visible in parser_report.errors.
Success result = normal extraction → sheet report is "parsed_ok" or "parsed_with_warnings", artifact data is assembled.
def _try_extract(self, sheet_name: str, extractor_class, result_class):
    """Three outcomes: None (missing), failed result (crashed), success result."""
    if not self._reader.has_sheet(sheet_name):
        return None  # → assembler treats as "skipped"

    try:
        return extractor_class(self._reader).extract()
    except Exception as e:
        import logging
        logging.getLogger(__name__).error(f"Pack 2 extraction failed for '{sheet_name}': {e}")

        error_issue = ParserIssue(
            code=f"{sheet_name.upper().replace(' ','_').replace('-','_')}_EXTRACTION_FAILED",
            severity="error", sheet_name=sheet_name,
            cell_ref=None, row_number=None, field_path=None,
            message=f"Extractor crashed: {e}",
            expected_value=None, actual_value=None, raw_value=None,
        )
        failed_report = SheetReport(
            sheet_name=sheet_name, status="failed",
            merged_regions_count=None, expected_body_range=None,
            rows_scanned=0, rows_kept=0, rows_skipped=0,
            warning_count=0, error_count=1,
        )
        return result_class(
            anchor_checks=[], issues=[error_issue], sheet_report=failed_report,
        )  # → assembler sees result exists but status="failed" → artifact null, error propagated

4.1 Assembler helper for result status

def _pack2_ok(result) -> bool:
    """True if result exists AND extraction succeeded (not failed)."""
    return result is not None and result.sheet_report.status != "failed"

def _pack2_has_report(result) -> bool:
    """True if result exists (even if failed) — has a sheet_report to propagate."""
    return result is not None

def run_cash_flow_cross_checks(cash_flow_identity, general_identity,
                                cash_flow_summary, bid_s_summary, top_sheet_summary):
    checks = []
    # 3 identity checks: inquiry_no, client_name, project_name vs General
    for fld in ["inquiry_no", "client_name", "project_name"]:
        checks.append(_exact_check(
            code=f"CASH_FLOW_vs_GENERAL_{fld.upper()}",
            left_field_path=f"cost_breakdown_profile.financial_profile.identity_mirror.{fld}",
            right_field_path=f"workbook_profile.rfq_identity.{fld}",
            left_value=_value(cash_flow_identity, fld),
            right_value=_value(general_identity, fld),
        ))
    # Inflow vs Bid S Grand Total (tight)
    checks.append(_numeric_check(
        code="CASH_FLOW_INFLOW_vs_BID_S_GRAND_TOTAL",
        left_field_path="cost_breakdown_profile.financial_profile.cash_flow_summary.total_inflow_sr",
        right_field_path="cost_breakdown_profile.bid_summary.grand_total.amount_sar",
        left_value=_value(cash_flow_summary, "total_inflow_sr"),
        right_value=_value(_value(bid_s_summary, "grand_total"), "amount_sar"),
        tolerance_abs=0.01,
    ))
    # Inflow vs Top Sheet Revenue (loose, informational)
    checks.append(_numeric_check(
        code="CASH_FLOW_INFLOW_vs_TOP_SHEET_REVENUE",
        left_field_path="cost_breakdown_profile.financial_profile.cash_flow_summary.total_inflow_sr",

        left_value=_value(cash_flow_summary, "total_inflow_sr"),
        right_value=_value(_value(top_sheet_summary, "total_revenue"), "rev00_value"),
        tolerance_abs=10.0, informational_only=True,
    ))
    return checks

def run_mat_breakup_cross_checks(mat_summary_total, bid_s_material_line, bid_s_meta, mat_items):
    checks = []
    # Total vs Bid S material SAR
    checks.append(_numeric_check(
        code="MAT_BREAKUP_TOTAL_vs_BID_S_MATERIAL",
        left_field_path="cost_breakdown_profile.material_decomposition.summary.grand_total.cost_total_sr",
        right_field_path="cost_breakdown_profile.bid_summary_lines[material].amount_sar",
        left_value=_value(mat_summary_total, "cost_total_sr"),
        right_value=_value(bid_s_material_line, "amount_sar"),
        tolerance_abs=0.01,
    ))
    # Finish weight vs Bid S weight
    checks.append(_numeric_check(
        code="MAT_BREAKUP_FINISH_WT_vs_BID_S_WEIGHT",
        left_field_path="cost_breakdown_profile.material_decomposition.summary.grand_total.weight_finish_ton",
        right_field_path="workbook_profile.bid_meta.total_weight_ton",
        left_value=_value(mat_summary_total, "weight_finish_ton"),
        right_value=_value(bid_s_meta, "total_weight_ton"),
        tolerance_abs=0.01,
    ))
    # Item sum vs summary
    item_cost_sum = sum(_value(_value(item, "grand_total"), "cost_total_sr") or 0 for item in mat_items)
    checks.append(_numeric_check(
        code="MAT_BREAKUP_ITEM_SUM_vs_SUMMARY",
        left_field_path="sum(material_decomposition.items[*].grand_total.cost_total_sr)",
        right_field_path="cost_breakdown_profile.material_decomposition.summary.grand_total.cost_total_sr",
        left_value=item_cost_sum,
        right_value=_value(mat_summary_total, "cost_total_sr"),
        tolerance_abs=0.01,
    ))
    return checks

6.4 B-O-Q cross-checks (per-item, variable count)

def run_boq_cross_checks(boq_item_details, general_item_rows):
    """Per-item weight: BOQ computed procured_weight_kg / 1000 vs General total_weight_ton."""
    checks = []
    for boq_item in boq_item_details:
        idx = _value(boq_item, "item_block_index")
        computed = _value(boq_item, "computed_total")
        boq_weight_kg = _value(computed, "procured_weight_kg") or 0
        if boq_weight_kg == 0 or idx is None:
            continue
        if idx <= len(general_item_rows):
            gen_item = general_item_rows[idx - 1]
            gen_weight_ton = _value(gen_item, "total_weight_ton")
            if gen_weight_ton and gen_weight_ton > 0:
                checks.append(_numeric_check(
                    code=f"BOQ_ITEM_{idx}_WEIGHT_vs_GENERAL",
                    left_field_path=f"boq_profile.boq_item_details[{idx}].computed_total.procured_weight_kg",
                    right_field_path=f"workbook_profile.general_item_rows[{idx}].total_weight_ton * 1000",
                    left_value=boq_weight_kg,
                    right_value=gen_weight_ton * 1000,
                    tolerance_rel=0.05,
                    informational_only=True,
                ))
    return checks

07 Orchestrator — parser_orchestrator.py

def parse(self, workbook_path, rfq_id, workbook_file_name=None, workbook_blob_path=None):
    workbook = Path(workbook_path)
    self._reader.open(workbook.as_posix())

    # Pack 1 (unchanged)
    match_result = TemplateMatcher(self._reader).validate()
    general_result = GeneralExtractor(self._reader).extract()
    bid_s_result = BidSExtractor(self._reader).extract()
    top_sheet_result = TopSheetExtractor(self._reader).extract()

    # Pack 2 (soft-fail: None=missing, failed result=crashed, success=normal)
    cash_flow_result = self._try_extract("Cash Flow", CashFlowExtractor, CashFlowExtractionResult)
    mat_breakup_result = self._try_extract("Mat Break-up", MatBreakupExtractor, MatBreakupExtractionResult)
    boq_result = self._try_extract("B-O-Q", BoqExtractor, BoqExtractionResult)

    # Cross-checks — only pass data for successful Pack 2 extractions
    def _p2_data(result):
        """Convert extraction result to dict for cross-checks, only if successful."""
        if result is None or result.sheet_report.status == "failed":
            return None
        return {k: v for k, v in result.__dict__.items() if k not in {"anchor_checks", "issues", "sheet_report"}}

    checks = run_cross_checks(
        general_data={"rfq_identity": general_result.rfq_identity,
                      "general_summary": general_result.general_summary},
        bid_s_data={"identity_mirror": bid_s_result.identity_mirror,
                    "bid_meta": bid_s_result.bid_meta,
                    "bid_summary_lines": bid_s_result.bid_summary_lines,
                    "bid_summary": bid_s_result.bid_summary},
        top_sheet_data={"identity_mirror": top_sheet_result.identity_mirror,
                        "top_sheet_lines": top_sheet_result.top_sheet_lines,
                        "top_sheet_summary": top_sheet_result.top_sheet_summary},
        cash_flow_data=_p2_data(cash_flow_result),
        mat_breakup_data=_p2_data(mat_breakup_result),
        boq_data=_p2_data(boq_result),
        general_item_rows=general_result.general_item_rows,
    )

    return build_envelope(
        rfq_id=rfq_id,
        workbook_file_name=workbook_file_name or workbook.name,
        workbook_blob_path=workbook_blob_path,
        workbook_format=workbook.suffix.lstrip(".").lower() or "xls",
        template_match=match_result.template_match,
        general_result=general_result,
        bid_s_result=bid_s_result,
        top_sheet_result=top_sheet_result,
        cash_flow_result=cash_flow_result,
        mat_breakup_result=mat_breakup_result,
        boq_result=boq_result,
        matcher_anchor_checks=match_result.anchor_checks,
        matcher_issues=match_result.issues,
        cross_checks=checks,
    )
